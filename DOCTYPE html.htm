        async function fetchImage(prompt) {

            const url = `https://generativelanguage.googleapis.com/v1beta/models/imagen-4.0-generate-001:predict?key=${apiKey}`;

            const payload = {

                instances: [{ prompt: `High-end food photography of ${prompt} ice cream, blue and pink pastel studio background, soft lighting, 8k resolution, appetizing` }],

                parameters: { sampleCount: 1 }

            };

            try {

                const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) });

                const result = await response.json();

                return `data:image/png;base64,${result.predictions[0].bytesBase64Encoded}`;

            } catch (error) { return null; }

        }

        function handleOrderClick(flavor) {

            if (!isLoggedIn) {

            } else {

                openOrderModal(flavor);

            }

        }

 

        function openOrderModal(flavor) {

            activeFlavor = flavor;

            document.getElementById('modal-title').innerText = flavor.name;

            document.getElementById('modal-price').innerText = `KSh ${flavor.price}`;

            document.getElementById('terms-check').checked = false;

           

            const cardImg = document.querySelector(`img[alt="${flavor.name}"]`);

            if (cardImg) document.getElementById('modal-image').style.backgroundImage = `url('${cardImg.src}')`;

           

            document.getElementById('order-modal').classList.remove('hidden');

        }

 

        function closeOrderModal() { document.getElementById('order-modal').classList.add('hidden'); }

 

        function confirmOrder(method) {

            const terms = document.getElementById('terms-check').checked;

            if (!terms) return alert("Please accept the Terms & Conditions to proceed.");

           

            const qty = document.getElementById('modal-qty').value;

            const total = activeFlavor.price * qty;

 

            if (method === 'whatsapp') {

                const text = `Hello B.E.G.I.N Ice Cream Kenya! 🍦%0A%0AOrder Details:%0AFlavor: *${activeFlavor.name}*%0AQuantity: *${qty}*%0ATotal: *KSh ${total}*%0A%0AI have read the T&Cs and will pay KSh ${total} to 0711446798 via M-Pesa.`;

                window.open(`https://wa.me/${phoneNumber}?text=${text}`, '_blank');

            } else {

                window.location.href = `tel:0711446798`;

            }

            closeOrderModal();

        }
